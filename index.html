Install the Auto Send Extension

One-time setup — takes under 2 minutes. Only needed for auto-send.
1
Download the extension
Then unzip it to a permanent folder (e.g. Desktop/wa-extension)
2
Open Chrome Extensions
Go to chrome://extensions in your browser and turn on Developer mode (top-right toggle)
3
Load the extension
Click Load unpacked → select the unzipped wa-extension folder
4
Log in to WhatsApp Web
Open web.whatsapp.com in the same Chrome window and scan the QR code once. Keep the tab open.
5
Done — click Start Sending!
Come back here, select contacts, type your message, and click Start Sending. The extension will press Send automatically for each contact.

/**
 * WhatsApp Bulk Messenger — Google Sheet Bridge
 * ================================================
 * Paste this into a Google Apps Script project bound to your Google Sheet.
 * Deploy as a Web App to connect your Contacts Directory.
 *
 * SETUP:
 * 1. Open your Google Sheet (or create a new one)
 * 2. First row should have headers: Name | Phone | Email | DOB | Group
 * 3. Extensions menu → Apps Script
 * 4. Delete any starter code, paste this entire file
 * 5. Click Deploy → New deployment → type: Web app
 *      - Execute as: Me
 *      - Who has access: Anyone
 * 6. Click Deploy, authorize when prompted
 * 7. Copy the Web App URL — paste it into the Bulk Messenger app
 */

const SHEET_NAME = ''; // leave blank to use the first/active sheet, or set a specific sheet name

function getSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  if (SHEET_NAME) {
    return ss.getSheetByName(SHEET_NAME) || ss.getActiveSheet();
  }
  return ss.getActiveSheet();
}

function jsonOut_(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * GET — returns all contacts as JSON
 * Used by the app to "Load from Sheet"
 */
function doGet(e) {
  try {
    const sheet = getSheet_();
    const data = sheet.getDataRange().getValues();
    if (data.length === 0) return jsonOut_({ success: true, contacts: [] });

    const headers = data[0].map(h => String(h).trim().toLowerCase());
    const idx = {
      name:  headers.indexOf('name'),
      phone: headers.indexOf('phone'),
      email: headers.indexOf('email'),
      dob:   headers.indexOf('dob'),
      group: headers.indexOf('group'),
    };

    const contacts = [];
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const name  = idx.name  >= 0 ? row[idx.name]  : '';
      const phone = idx.phone >= 0 ? row[idx.phone] : '';
      if (!name && !phone) continue; // skip blank rows

      contacts.push({
        name:  String(name || ''),
        phone: String(phone || ''),
        email: idx.email >= 0 ? String(row[idx.email] || '') : '',
        dob:   idx.dob   >= 0 ? formatDob_(row[idx.dob]) : '',
        group: idx.group >= 0 ? String(row[idx.group] || '') : '',
      });
    }

    return jsonOut_({ success: true, contacts: contacts });
  } catch (err) {
    return jsonOut_({ success: false, error: err.message });
  }
}

function formatDob_(val) {
  if (!val) return '';
  if (val instanceof Date) {
    return Utilities.formatDate(val, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  return String(val);
}

/**
 * POST — receives an action from the app:
 *   { type: 'sync_all', contacts: [...] }   → overwrites the whole sheet
 *   { type: 'add', contact: {...} }          → appends one row
 *   { type: 'delete', phone: '...' }         → deletes row matching phone
 *
 * NOTE: app sends Content-Type: text/plain to avoid CORS preflight issues —
 * this script still parses it as JSON.
 */
function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const sheet = getSheet_();

    if (body.type === 'sync_all') {
      sheet.clear();
      sheet.appendRow(['Name', 'Phone', 'Email', 'DOB', 'Group']);
      const rows = (body.contacts || []).map(c => [
        c.name || '', c.phone || '', c.email || '', c.dob || '', c.group || ''
      ]);
      if (rows.length > 0) {
        sheet.getRange(2, 1, rows.length, 5).setValues(rows);
      }
      return jsonOut_({ success: true, count: rows.length });
    }

    if (body.type === 'add') {
      const c = body.contact || {};
      sheet.appendRow([c.name || '', c.phone || '', c.email || '', c.dob || '', c.group || '']);
      return jsonOut_({ success: true });
    }

    if (body.type === 'delete') {
      const data = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][1]).trim() === String(body.phone).trim()) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
      return jsonOut_({ success: true });
    }

    return jsonOut_({ success: false, error: 'Unknown action type' });
  } catch (err) {
    return jsonOut_({ success: false, error: err.message });
  }
}
